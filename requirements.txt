#  NVIDIA (CUDA 12.1):

#  NVIDIA (CUDA 11.8, старые карты GTX 10xx / 16xx):
#    pip install torch torchaudio --index-url https://download.pytorch.org/whl/cu118
#
#  AMD Radeon (ROCm 6.0, Linux):
#    pip install torch torchaudio --index-url https://download.pytorch.org/whl/rocm6.0
#
#  AMD Radeon (ROCm 5.7, старые карты RX 5xxx/6xxx):
#    pip install torch torchaudio --index-url https://download.pytorch.org/whl/rocm5.7
#
#  CPU (любое железо, медленнее):
#    pip install torch torchaudio --index-url https://download.pytorch.org/whl/cpu
